@startuml
skinparam classAttributeIconSize 0
skinparam monochrome true

package "GRAPP Domain Model" {

    class Souradnice <<Value Object>> {
        public latitude: double
        public longitude: double
    }

    class Mapa {
        private mapovyPodklad: String
        private aktualniZoom: int
        public vykresliObjekty(): void
        public aplikujFiltr(f: Filtr): void
        public obnovData(intervalMs: int): void
    }

    class Vlak {
        private idVlaku: String
        private cisloVlaku: String
        private nazevVlaku: String
        private /zpozdeniMinuty: int <<derived>>
        private posledniAktualizace: DateTime
        public ziskejAktualniZpozdeni(): int
        public zobrazDetail(): void
    }

    class TypVlaku <<Enumeration>> {
        private kod: String
        private popis: String
        private barvaIkony: String
    }

    class Dopravce {
        private kodDopravce: String
        private nazev: String
        private barvaHex: String
    }

    class Trasa {
        private idTrasy: String
        private delkaKm: float
        private platiOd: Date
        private aktivni: boolean
        public ziskejVychoziStanici(): Stanice
        public ziskejCilovouStanici(): Stanice
    }

    class Zastavka {
        private pravidelnyPrijezd: DateTime
        private pravidelnyOdjezd: DateTime
        private ocekavanyPrijezd: DateTime
        private ocekavanyOdjezd: DateTime
        private kolej: String
        private nastupiste: String
        private projizdi: boolean
        public jeZpozdena(): boolean
    }

    class Stanice {
        private sr70: int
        private nazev: String
        public generujOdjezdovouTabuli(): List<Zastavka>
    }

    class Filtr {
        private hledaneCislo: String
        private skrytaZpozdeniDo: int
        public pridejDopravce(d: Dopravce): void
        public pridejTypVlaku(t: TypVlaku): void
        public splnujePodminky(v: Vlak): boolean
    }

    ' Vztahy
    Mapa "1" *-- "0..*" Vlak : zobrazuje
    Mapa "1" *-- "0..*" Stanice : zobrazuje
    Mapa "1" o-- "1" Filtr : používá

    Mapa *-- Souradnice : stredMapy
    Vlak *-- Souradnice : poloha
    Stanice *-- Souradnice : poloha

    Vlak "N" -- "1" Dopravce : je provozován >
    Vlak "1" -- "1" TypVlaku : je klasifikován jako >
    Vlak "1" *-- "1" Trasa : projíždí >

    Trasa "1" *-- "2..*" Zastavka : skládá se z >
    Zastavka "*" -- "1" Stanice : fyzicky leží v >

}
@enduml
